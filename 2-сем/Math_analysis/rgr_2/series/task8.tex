\Problem{Розвинути функцію
$
\frac{x^2}{\sqrt[3]{1+x^2}}
$
в ряд Маклорена та вказати
його область збіжності}

	% Формула Маклорена:
% \begin{dmath}
	% f(x) = f(0) + \frac{f'(0)}{1!}x + \frac{f''(0)}{2!}x^2 +\cdots+
	% \frac{f^{(n)}(0)}{n!}x^n + r_n(x), \\
	% r_n(x) = \frac{f^{(n+1)}(\Theta)}{(n+1)!}x^{n+1}
	% \label{maclaurin}
% \end{dmath}

\Solution

\begin{align}
	f' = \frac{2\,x}{\sqrt[{3}]{{x}^{2}+1}}-\dfrac{2\,{x}^{3}}{3\,\left({{x}^{2}+1}\right)^{\frac{4}{3}}} = 0 \\
% (2x)/sqrt(1+x^2)
	f'' =
		\frac{4\,{x}^{4}+6\,{x}^{2}+18}{\sqrt[{3}]{{x}^{2}+1}\,\left(9\,{x}^{4}+18\,{x}^{2}+9\right)} = \frac{18}{9} = 2
		\\
	f''' = -\frac{8\,{x}^{5}+216\,x}{\sqrt[{3}]{{x}^{2}+1}\,\left(27\,{x}^{6}+81\,{x}^{4}+81\,{x}^{2}+27\right)} = 0
		\\
	f^V =
		-\frac{320\,{x}^{7}-2400\,{x}^{5}+74880\,{x}^{3}-38880\,x}{\sqrt[{3}]{{x}^{2}+1}\,\left(243\,{x}^{10}+1215\,{x}^{8}+2430\,{x}^{6}+2430\,{x}^{4}+1215\,{x}^{2}+243\right)}
		\\
	f^{IV}(0) =
	\frac{40\,{x}^{6}-120\,{x}^{4}+3672\,{x}^{2}-648}{\sqrt[{3}]{{x}^{2}+1}\,\left(81\,{x}^{8}+324\,{x}^{6}+486\,{x}^{4}+324\,{x}^{2}+81\right)}
	= -8 \\
	f^{VI} =
		% \frac{3520\,{x}^{8}-47520\,{x}^{6}+1758240\,{x}^{4}-1801440\,{x}^{2}+116640}{\sqrt[{3}]{{x}^{2}+1}\,\left(729\,{x}^{12}+4374\,{x}^{10}+10935\,{x}^{8}+14580\,{x}^{6}+10935\,{x}^{4}+4374\,{x}^{2}+729\right)}
		160
\end{align}
% (4*x^5+120*x^3-108*x)/(sqrt3(x^2+1)*(9*x^6+27*x^4+27*x^2+9))
Отримуємо ряд:
\begin{equation}
	\frac{x^2}{\sqrt[3]{1+x^2}}
	= 0+0x+\frac{2x^2}{2!}+0-\frac{8x^4}{4!} + \cdots
	= x^2-\frac{x^4}{3} %+\frac{160x^6}{6!} =
		+ \frac{2x^6}{9}
		+ \cdots
		+ \frac{(-1)^{n+1}x^{2n}}{3^{n-1}}
\end{equation}

За ознакою д'Аламбера й теоремою Абеля:
\begin{equation}
	\lim_{n\to\infty}
	\left|
		\frac{3^{n-1} \cdot (-1)^{n+2} \cdot x^{2(n+1)}}
		{(-1)^{n+1}\cdot x^{2n}\cdot 3^n}
	\right|
	=
	\left|
		\frac{x^2}{-3}
	\right| < 1
\end{equation}
Область збіжності ряду: $x\in (-\sqrt3;\sqrt3)$
% $$
% x^2-\frac{x^4}{3}+\frac{2x^6}{9}-\frac{14x^8}{81}+O(x^9)
% $$


\Answer{
	$
	\sum_{n=1}^\infty \frac{(-1)^{n+1}x^{2n}}{3^{n-1}}
	$;
	ОЗ: $x\in (-\sqrt3;\sqrt3)$
}
