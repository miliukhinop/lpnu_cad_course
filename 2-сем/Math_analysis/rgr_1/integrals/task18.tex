% 1/(x(x^3+1)^(1/4) <++>
\Problem{
	Знайти інтеграл від біномного диференціала
	$
	\int \frac{ dx }{ x\sqrt[4]{x^3 + 1} }
	$
}
\Solution

\begin{equation}
	% \int \frac{ dx }{ x\sqrt[4]{x^3 + 1} }
	\int{\dfrac{1}{x\,\sqrt[{4}]{{x}^{3}+1}}}{\;\mathrm{d}x}
	=\int x^{-1}(x^3+1)^{-\frac{1}{4}}\,dx
\end{equation}

\begin{enumerate}
	\item $-\frac{1}{4}$ не ціле
	\item $\frac{-1+1}{3}$ ціле $\implies$ заміна:
	\fbox
	{
		$\begin{array}{c|c}
			1 + x^3 = t^4 & x^3 = t^4-1 \\
			x^{-1}=\frac{1}{\sqrt[3]{t^4-1}} &
			dx = \frac{1}{3}(t^4-1)^{-\frac{2}{3}}*4t^3\, dt
		\end{array}$
	}
\end{enumerate}

\begin{dmath}
	\int(t^4-1)^{-\frac{1}{3}}*t^4\,dt=
	\int(t^4-1)^{-\frac{1}{3}}*t^4\,dt
\end{dmath}

\Answer{
	$
\frac{\ln\left(\left|\sqrt[{4}]{{x}^{3}+1}-1\right|\right)}{3}-\frac{\ln\left(\sqrt[{4}]{{x}^{3}+1}+1\right)}{3}+\frac{2\,\arctan\left(\sqrt[{4}]{{x}^{3}+1}\right)}{3}+C
	$
}
