\Problem{
	$
	\int \frac{\cos x}{\sqrt{(\sin x-4)^3}} \, dx
	$
}
\Solution

\begin{dmath}
	\int \frac{\cos x}{\sqrt{(\sin x-4)^3}} \, dx =
	\frac{1}{3} \int \frac{\cancel{\cos x}}{\cancel{\cos x}\sqrt{(\sin x-4)^3}} \, d(\underbrace{\sin x-4}_t) =
	\int t^{-\frac{3}{2}} \, dt =
	-\frac{2}{\sqrt t} =
	-\frac{2}{\sqrt{\sin x-4}}
\end{dmath}

\Answer{
	$
	-\frac{2}{\sqrt{\sin x-4}}
	$
}
