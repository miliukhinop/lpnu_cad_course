\Problem{Знайти невизначений інтеграл
	$\int \frac{dx}{ \sqrt{5x^2+3} }$
}
\Solution

\begin{equation}
	\int \frac{dx}{ \sqrt{5x^2+3} } =
	\frac{1}{\sqrt{5}}\int \frac{d\sqrt5 x}{ \sqrt{5x^2+3} } =
	\frac{1}{\sqrt{5}}\ln\left|\sqrt5 x + \sqrt{5x^2+3}\right|
\end{equation}

\Answer{
	$
	\frac{1}{\sqrt{5}}\ln\left|\sqrt5 x + \sqrt{5x^2+3}\right|
	$
}
