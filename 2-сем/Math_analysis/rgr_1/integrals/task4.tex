\Problem{
	$\int e^{5x^2-3} x\, dx$
}
\Solution

\begin{equation}
	\int e^{5x^2-3} x\, dx =
	\frac{1}{10} \int e^{5x^2-3} \, d(5x^2-3) =
	\frac{1}{10} e^{5x^2-3} + C
\end{equation}

\Answer{
	$
	\frac{1}{10} e^{5x^2-3} + C
	$

}
