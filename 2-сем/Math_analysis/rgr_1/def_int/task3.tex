\begin{dmath}
	S = \int_0^2 \sqrt{x}\,dx + \int_2^4 \sqrt{x}-(x-2) \,dx
	=  \int_0^2 \sqrt{x}\,dx + \int_2^4 \sqrt{x}\,dx - \int_2^4 x-2 \,dx
	= \frac{2\sqrt{x^3}}{3} \Bigg|^2_0 + \frac{x^{\frac{3}{2}}}{\frac{3}{2}} \Bigg|^4_2
	- \left( \frac{x^2}{2} -2x \right) \Bigg|^4_2
	= \frac{2\sqrt{8}}{3} + \frac{2\sqrt{4^3}}{3} - \frac{2\sqrt{8}}{3} - \left( \frac{16}{2} - 8 -\frac{4}{2} + 4 \right) = \frac{16}{3} - 2 = \frac{10}{3}
\end{dmath}
