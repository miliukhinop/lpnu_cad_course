\Problem{Знайти невизначений інтеграл
	$\int \frac{dx}{ \sqrt{4-7x^2} }$
}
\Solution

\begin{equation}
	\int \frac{dx}{ \sqrt{4-7x^2} } =
	\frac{1}{\sqrt{7}} \int \frac{dx}{ \sqrt{ \frac{4}{7}-x^2} } =
	\frac{1}{\sqrt{7}} \arcsin\frac{ x\sqrt{7} }{2}
\end{equation}

\Answer{
	$
	\frac{1}{\sqrt{7}} \arcsin\frac{ x\sqrt{7} }{2}
	$
}
