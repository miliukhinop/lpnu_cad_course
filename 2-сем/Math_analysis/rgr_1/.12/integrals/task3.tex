\Problem{
	$\int \frac{\sin 3x}{\cos^2 3x} \, dx$
}
\Solution

\begin{dmath}
	\int \frac{\sin 3x}{\cos^2 3x} \, dx =
	\frac{1}{3} \int \frac{\sin 3x}{\cos^2 3x} \, d(3x) =
	- \frac{1}{3} \int \frac{1}{\cos^2 3x} \, d( \underbrace{\cos3x}_t ) =
	- \frac{1}{3} \int \frac{1}{t^2} \, dt =
	- \frac{1}{3} \cdot \frac{-1}{t} + C =
	\frac{1}{3\cos3x} + C
\end{dmath}

\Answer{
	$\displaystyle\frac{1}{3\cos3x} + C$
}
