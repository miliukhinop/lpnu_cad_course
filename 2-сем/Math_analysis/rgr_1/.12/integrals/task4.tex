\Problem{
	$\int e^{1 - 4x^2} x\, dx$
}
\Solution

\begin{equation}
	\int e^{1 - 4x^2} x\, dx =
	- \frac{1}{8} \int e^{1 - 4x^2} \, d(1-4x^2) =
	- \frac{1}{8} e^{1-4x^2} + C
\end{equation}

\Answer{
	$
	- \frac{1}{8} e^{1-4x^2} + C
	$

}
