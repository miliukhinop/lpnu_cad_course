\Problem{Вказати хоча б одну нескінченно малу величину, еквівалентну даній }

\paragraph{а) $(x+h)^3 - x^3 (h\to 0)$}

\Solution
\begin{equation}
	(x+h)^3 - x^3 =
	0
\end{equation}

\paragraph{б) $\ln\frac{1+x}{1-x} (x\to 0)$}

\Solution

\begin{equation}
	\ln\frac{1+x}{1-x} (x\to 0) = \ln{(1+x)} - \ln{(1-x)} = x - \ln{(1-x)}
\end{equation}

\begin{equation}
	\lim_{x\to0}{x - \ln{(1-x)}} = 0 - \ln1 = 0.
\end{equation}

\begin{equation}
	\lim_{x\to0}\ln\frac{1+x}{1-x} =
	\ln\frac{1}{1} = 0
\end{equation}
