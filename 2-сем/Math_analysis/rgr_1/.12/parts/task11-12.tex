\Problem{Використовуючи диференціал, знайти $\arcsin0.48$
}

\Solution

	%\small
\begin{align}
f(x_0+\Delta x) \approx f(x_0) + f'(x_0)\Delta  x\\
	f(x) = \arcsin x,~
	x_0 = 0.5, \Delta x = -0.02 \\
	f'(\arcsin x) = \frac{1}{\sqrt{1 - x^2}},~
	f'(x_0) = \frac{1}{\sqrt{1 - 0.5^2}}
	= \frac{1}{\sqrt{0.75}} \\
	\arcsin(0.48) = \arcsin{\frac{1}{2}} + \frac{1}{\sqrt{0.75}} * (-0.02) =
	%\frac{\pi}{6} + \frac{2}{\sqrt{3}} * (-\frac{2}{100}) =
	\frac{\pi}{6} + \frac{2}{50\sqrt{3}}; \\
	\sqrt{3} = \sqrt{4 - 1} = \sqrt{4} - \frac{1}{2\sqrt{4}} * (-1) =
	2\frac{1}{4} \\ % ??? <++>
	\arcsin(0.48) \approx \frac{\pi}{6} + \frac{9}{4} =
	\frac{2\pi + 27}{12}.
	%f'(x^{1/2}) = 1/2 * x^{-1/2}
\end{align}

\Answer{$\frac{2\pi + 27}{12}.$}

\Problem{
	Написати рівняння дотичної і нормалі до графіка функції
	$x = t^2$, $y = t^3$ у точці $M(4,8)$
}

\Solution

\begin{equation}
\begin{cases}
	x = t^2 \to x'_t = 2t \\
	y = t^3 \to y'_t = 3t^2 \\
\end{cases}
\end{equation}
\begin{dmath}
%	x = y^{\frac{2}{3}}, y \geq 0
	%x'_t = 2t, y'_t = 3t^2 \\
	y'_x = \frac{y'_t}{x'_t} = \frac{2}{3t};~
	4 = t^2, 8 = t^3 \to t = 2
	%t^2 = (t^3)^{\frac{2}{3}}
\end{dmath}

\paragraph{Рівняння дотичної:}
\begin{equation}
	y - y_0 = f'(x_0)(x - x_0),
	y - 8 = \frac{1}{3}(x - 4)
\end{equation}

\paragraph{Рівняння нормалі:}
\begin{equation}
	y - y_0 = -\frac{1}{f'(x_0)}(x - x_0),
	y - 8 = -\frac{3}{1}(x - 4)
\end{equation}

\Answer{
	Дотична: $y - 8 = \frac{1}{3}(x - 4)$.
	{Нормаль: $y - 8 = -\frac{3}{1}(x - 4)$}
}
